class Solution {
public:
    int mySqrt(int x)
    {
        if(x<2){
            return x;
        }
        long long int i=2,j=x/2;
        
        while(i<=j)
        {
            long long int mid=i+(j-i)/2;
            if(mid*mid==x||(mid*mid<x&&(mid+1)*(mid+1)>x)){
                return mid;
            }
            else if(mid*mid>x){
                j=mid-1;
            }
            else{
                i=mid+1;
            }
        }
        return j;
    }
};
